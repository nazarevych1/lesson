upper_wick_length = high - max(open, close)

Entry rules (bullish):
1. lower_wick_length >= self.WICK_RATIO * body and
2. close > open_ and
3. upper_wick_length * self.OP_WICK_RATIO <= lower_wick_length

stop_loss = low
take_profit = close + (close - low) * self.RR


Entry rules (bearish):
1. upper_wick_length >= self.WICK_RATIO * body and
2. close < open_ and
3. lower_wick_length * self.OP_WICK_RATIO <= upper_wick_length

stop_loss = high
